        function formatPrice(price) {

            if (price >= 1000) return price.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });

            if (price >= 1) return price.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 4 });

            if (price >= 0.01) return price.toLocaleString('en-US', { minimumFractionDigits: 4, maximumFractionDigits: 6 });

        }

        // --- Загрузка данных с CoinGecko ---

        async function fetchCryptoData() {

            showLoadingSkeletons('mainCoins');

            showLoadingSkeletons('memeCoins');


            try {

                const response = await fetch(`${COINGECKO_API_MARKETS}?vs_currency=usd&ids=${[...MAIN_COINS_IDS, ...MEME_COINS_IDS].join(',')}&price_change_percentage=1h,24h`);

                if (!response.ok) {

                    throw new Error(`HTTP error! status: ${response.status}`);

                }

                const data = await response.json();


                cryptoData.main = data.filter(coin => MAIN_COINS_IDS.includes(coin.id));

                cryptoData.meme = data.filter(coin => MEME_COINS_IDS.includes(coin.id));

                

                renderCoins(cryptoData.main, 'mainCoins');

                renderCoins(cryptoData.meme, 'memeCoins', true);


            } catch (error) {

                console.error('Ошибка при получении данных с CoinGecko:', error);

                showError('mainCoins', 'Не удалось загрузить основные данные.');

                showError('memeCoins', 'Не удалось загрузить данные мем-коинов.');

            }

        }


        // --- Обработка событий ---

        function handleTouchStart(e) {

            touchStartX = e.touches[0].clientX;

        }


        function handleTouchEnd(e) {

            touchEndX = e.changedTouches[0].clientX;

            handleSwipe();

        }


        function handleSwipe() {

            const swipeThreshold = 50;

            const diff = touchStartX - touchEndX;


            if (Math.abs(diff) > swipeThreshold) {

                if (diff > 0 && currentTab < 2) {

                    // Свайп влево

                    switchTab(currentTab + 1);

                } else if (diff < 0 && currentTab > 0) {

                    // Свайп вправо

                    switchTab(currentTab - 1);

                }

            }

        }


        function switchTab(index) {

            currentTab = index;

            

            // Обновляем индикатор

            document.getElementById('tabIndicator').className = `tab-indicator tab-${index}`;

            

            // Обновляем активную вкладку

            document.querySelectorAll('.mobile-tab').forEach((tab, i) => {

                tab.classList.toggle('active', i === index);

            });


            // Обновляем активный элемент нижней навигации

            document.querySelectorAll('.bottom-nav .nav-item').forEach((item, i) => {

                item.classList.toggle('active', i === index); // предполагаем, что первые 3 элемента bottom-nav соответствуют вкладкам

            });

            

            // Перемещаем контейнер

            const container = document.getElementById('swipeContainer');

            container.style.transform = `translateX(-${index * 33.33}%)`;


            // Обновляем контент вкладки при переключении (если требуется)

            // fetchCryptoData(); // Можно вызывать здесь, но сейчас это происходит по интервалу

        }

            showLoadingSkeletons('memeCoins');

            container.addEventListener('touchstart', handleTouchStart);

        }

        init();
